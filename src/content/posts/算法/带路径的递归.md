## 字符串的全部子序列
https://www.nowcoder.com/practice/92e6247998294f2c933906fdedbc6e6a

> 带路径的递归

```cpp
// 生成所有子序列（去重后）的函数
vector<string> generatePermutation(string s) {
    // 使用 set 去重（因为可能有重复字符）
    unordered_set<string> set;
    string path = "";   // path 用来存放当前递归选择的子序列
    f1(s, 0, path, set); // 从下标 0 开始递归
    vector<string> ans;
    // 将 set 中的结果转存到 vector 返回
    for(auto cur : set){
        ans.push_back(cur);
    }
    return ans;
}

// 递归函数：生成所有子序列
// 参数说明：
//   s   ：输入的字符串
//   i   ：当前递归到 s 的第几个字符
//   path：当前构造的子序列（不断扩展或回溯）
//   set ：存放结果，自动去重
void f1(string s, int i, string& path, unordered_set<string>& set){
    // 递归终止条件：已经遍历完字符串 s
    if(i == (int)s.size()){
        set.insert(path); // 把当前构造的子序列加入结果
    }else{
        // 选择：把 s[i] 放入 path
        path += s[i];
        f1(s, i + 1, path, set);   // 递归到下一个字符
        // 撤销选择（回溯）
        path.erase(path.size() - 1);

        // 不选择：跳过 s[i]
        f1(s, i + 1, path, set);
    }
}
```

## 90. 子集 II - 力扣
https://leetcode.cn/problems/subsets-ii/description/

```cpp
class Solution {
public:
    vector<vector<int>> subsetsWithDup(vector<int>& nums) {
        vector<vector<int>> ans;
        vector<int> path(nums.size());
        sort(nums.begin(), nums.end());
        f(nums ,0, path, 0, ans);

        return ans;
    }
    //i遍历数组
    void f(vector<int>& nums,int i,vector<int>& path, int size, vector<vector<int>>& ans){
        if(i == nums.size()){
            vector<int> cur(path.begin(), path.begin() + size);
            ans.push_back(cur);
        }else{
            int j = i + 1;
            while(j < nums.size() && nums[j] == nums[i]){
                j++;
            }
            f(nums, j, path, size, ans);
            for(; i < j; i++){
                path[size++] = nums[i];
                f(nums, j, path,size, ans);
            }
        }
    }
};
```

## 46. 全排列 - 力扣
https://leetcode.cn/problems/permutations/description/

```cpp
class Solution {
public:
    vector<vector<int>> permute(vector<int>& nums) {
        vector<vector<int>> ans;
        f(nums, 0, ans);
        return ans;
    }

    void f(vector<int>& nums, int i, vector<vector<int>>& ans){
        if(i == nums.size()){
            vector<int> cur(nums.begin(), nums.end()); 
            ans.push_back(cur);
        }else{
            for(int j = i; j < nums.size(); j++){
                swap(nums, i, j);
                f(nums, i + 1,ans);
                swap(nums, i, j);
            }
        }
    }
    void swap(vector<int>& nums, int i, int j){
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
};
```

```cpp
        return ans;
    }

        if(i == nums.size()){
            vector<int> cur(nums.begin(), nums.end());

            unordered_set<int> set;
            for(int j = i; j < nums.size(); j++){

                    swap(nums, i, j);
                }
                
            }
        }
    }

        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
};
```

## 用递归逆序一个栈
> 需要两个函数来实现
```cpp
// 功能：取出栈底元素，并保持其余元素的相对顺序不变
// 参数：stack<int> stack —— 注意这里是按值传递（副本），
//       如果要真正修改原栈，应使用引用 stack<int>&
int bottomOut(stack<int>& stack) {
    int ans = stack.top();  // 取出当前栈顶
    stack.pop();            // 弹出栈顶
    if (stack.empty()) {    // 如果此时栈空了，说明 ans 是原来的栈底
        return ans;         // 返回栈底元素
    } else {
        // 递归调用，继续向下寻找栈底
        int last = bottomOut(stack);
        // 把之前弹出的元素再压回去（恢复栈的状态）
        stack.push(ans);
        return last;        // 返回最底部的元素
    }
}

// 功能：递归反转栈
// 思路：每次取出栈底元素 -> 递归反转剩余栈 -> 再把取出的栈底元素放到栈顶
void reverse(stack<int>& stack) {
    if (stack.empty()) {    // 空栈，无需处理
        return;
    }
    // 取出栈底元素
    int num = bottomOut(stack);
    // 递归反转剩余栈
    reverse(stack);
    // 将原来的栈底元素压入栈顶
    stack.push(num);
}

```

## 用递归排序一个栈.

实现过程.
1. 求出深度
2. 已知深度,求出最大值
3. 已知深度和最大值,求出最大值出现次数
4. 知道深度,最大值,出现次数,让最大值沉底.
5. 不断循环2、3、4方法.

```cpp
// 主函数：对栈进行排序
void sort(stack<int> stack) {
    int h = deep(stack);  // 获取栈的高度（元素个数）
    while (h > 0) {
        int max = maxnum(stack, h);     // 找到前 h 个元素中的最大值
        int k = times(stack, h, max);   // 统计最大值出现的次数
        down(stack, h, max, k);         // 将最大值放到底部
        h -= k;                         // 减少剩余未排序的元素个数
    }
}

// 递归计算栈的深度（元素个数）
int deep(stack<int>& stack){
    if(stack.empty()){
        return 0;
    }
    int num = stack.top();  // 取出栈顶元素
    stack.pop();            // 弹出栈顶
    int d = deep(stack) + 1;// 递归计算剩余元素个数
    stack.push(num);        // 恢复栈顶元素
    return d;
}

// 在栈的前 h 个元素中找到最大值
int maxnum(stack<int>& stack, int h){
    if(h == 0) {
        return INT_MIN; // 没有元素时返回最小整数
    }
    int num = stack.top();  // 取出栈顶
    stack.pop();            // 弹出栈顶
    int restMax = maxnum(stack, h - 1); // 递归查找剩余元素的最大值
    int mnum = max(num, restMax);       // 当前元素与剩余最大值比较
    stack.push(num);        // 恢复栈顶
    return mnum;
}

// 统计最大值在前 h 个元素中出现的次数
int times(stack<int>& stack, int h, int max){
    if(h == 0){
        return 0;
    }
    int num = stack.top();  // 取出栈顶
    stack.pop();            // 弹出
    int restTimes = times(stack, h - 1, max); // 递归统计剩余次数
    int times = restTimes + (num == max ? 1 : 0); // 当前是否等于最大值
    stack.push(num);        // 恢复栈顶
    return times;
}

// 将最大值“沉底”操作（类似选择排序，每次把最大值放到栈底）
void down(stack<int>& stack, int h, int max, int k){
    if(h == 0){
        // 当栈为空时，把最大值入栈 k 次
        for(int i = 0; i < k; i++){
            stack.push(max);
        }
    }else{
        int num = stack.top();  // 取出栈顶元素
        stack.pop();            // 弹出
        down(stack, h - 1, max, k); // 递归处理剩余元素
        if(num != max){         // 如果当前元素不是最大值，就重新压回栈
            stack.push(num);
        }
    }
}

```

## 汉诺塔问题

```cpp

// 汉诺塔主函数，n 表示圆盘数量
void hanoi(int n) {
    if (n > 0) {
        f(n, "左", "右", "中");
    }
}

// 递归函数：把 i 个盘子从 from 移到 to，other 是辅助柱子
void f(int i, string from, string to, string other) {
    if (i == 1) {
        // 递归出口：只剩一个盘子，直接移动
        cout << "移动圆盘 1 从 " << from << " 到 " << to << endl;
    } else {
        // 1. 先把 i-1 个盘子从 from 移到 other
        f(i - 1, from, other, to);

        // 2. 把第 i 个盘子从 from 移到 to
        cout << "移动圆盘 " << i << " 从 " << from << " 到 " << to << endl;

        // 3. 再把 i-1 个盘子从 other 移到 to
        f(i - 1, other, to, from);
    }
}
```
