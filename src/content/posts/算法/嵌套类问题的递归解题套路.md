解题套路:
1. 定义全局变量 int where
2. 递归函数f(i) : s\[i..]，从i位置出发开始解析，遇到 字符串终止 或 嵌套条件终止 就返回
3. 返回值是f(i)负责这一段的结果
4. f(i)在返回前更新全局变量where，让上级函数通过where知道解析到了什么位置，进而继续

执行细节:
1. 如果f(i)遇到 嵌套条件开始，就调用下级递归去处理嵌套，下级会负责嵌套部分的计算结果
2. f(i)下级处理完成后，f(i)可以根据下级更新的全局变量where，知道该从什么位置继续

## 表达式求值

https://www.nowcoder.com/practice/c215ba61c8b1443b996351df929dc4d4

是用递归和两个栈实现,一个数字栈,一个符号栈
函数内维护一个cur变量,表示现在位置上的数字,如果符号栈顶是`*`或`/`,两栈同时弹出,进行运算后再压入栈.遇到`+`或者`-`就把数字和符号压入栈.
遇到左括号就进递归,遇到右括号就返回当前两栈的计算值,并更新走到的位置.

```cpp

int where;
int compute(vector<int> numbers, vector<char> ops);

void push(vector<int>& numbers, vector<char>& ops, int cur, char op) {
    int n = numbers.size();
    if(n == 0 || ops[n-1] == '+' || ops[n-1] == '-') {
        numbers.push_back(cur);
        ops.push_back(op);
    } else {
        // 遇到乘除，计算后替换栈顶
        int topNumber = numbers[n-1];
        char topOp = ops[n-1];
        if(topOp == '*') {
            numbers[n-1] = topNumber * cur;
        } else {
            numbers[n-1] = topNumber / cur;
        }
        ops[n-1] = op;
    }
}

// s[i.....] 开始计算, 遇到字符串终止 或者 遇到 ) 停止
// 返回 : 自己负责的这一段计算的结果.
// 返回之前要更新全局变量 where
int f(const char* s, int i) {
    int cur = 0;
    vector<int> numbers; // 数字栈
    vector<char> ops;    // 符号栈
    int n = strlen(s);

    while(i < n && s[i] != ')') {
        if(s[i] >= '0' && s[i] <= '9') {
            // 遇到数字
            cur = cur * 10 + (s[i++] - '0');
        } else if(s[i] != '(') {
            // 遇到了运算符
            push(numbers, ops, cur, s[i++]);
            cur = 0;
        } else {
            // 遇到括号，递归处理
            cur = f(s, i + 1);
            i = where + 1;
        }
    }
    // 把最后一个数推入
    push(numbers, ops, cur , '+'); 
    where = i; // 更新位置
    return compute(numbers, ops);
}

int compute(vector<int> numbers, vector<char> ops) {
    int n = numbers.size();
    int ans = numbers[0];
    for(int i = 1; i < n; i++) {
        ans += ops[i-1] == '+' ? numbers[i] : -numbers[i];
    }
    return ans;
}

int solve(string s) {
    where = 0;
    return f(s.c_str(), 0);
}
```


## 含有嵌套的字符串解码
https://leetcode.cn/problems/decode-string/

```cpp
class Solution {
public:
    int where;
    string decodeString(string s) {
        where = 0;
        return f(s.c_str(), 0);
    }

    string f(const char* s, int i){
        int cnt = 0;
        string ans;
        int n = strlen(s);
        while(i < n && s[i] != ']'){
            if((s[i] >= 'a' && s[i] <= 'z') || (s[i] >= 'A' && s[i] <= 'Z')){
                // 普通字母直接加进去
                ans.push_back(s[i++]);
            }else if(s[i] >= '0' && s[i] <= '9'){
                // 数字解析成次数
                cnt = cnt * 10 + (s[i++] - '0');
            }else if(s[i] == '['){
                // 遇到左括号，递归求子串
                string sub = f(s, i + 1);
                for(int k = 0; k < cnt; k++) {
                    ans += sub; // 拼接 cnt 次
                }
                cnt = 0;          // 用完清零
                i = where + 1;    // 跳过右括号
            }
        }
        where = i; // 更新当前位置
        return ans;
    }
};

```

## 嵌套分子式求原子数量

遍历的时候我们会遇到:大写,小写,数组,和左括号
遇到大写和左括号说明:历史记录需要结算,新的记录将要开始

有一个总的map来记录所有的个数
需要三种记录方式:
1. cnt 记录出现的个数
2. name 记录当前的字母
3. 表: 记录括号内的各种字母.
```cpp
#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    int where;

    string countOfAtoms(string str) {
        where = 0;
        map<string, int> mp = f(str.c_str(), 0);
        string ans;
        for (auto &p : mp) {
            ans += p.first;
            if (p.second > 1) {
                ans += to_string(p.second);
            }
        }
        return ans;
    }

private:
    // s[i...] 开始计算，遇到字符串终止 或者 遇到 ')' 停止
    // 返回 : 当前子串的结果 map
    // 返回之前更新全局变量 where
    map<string, int> f(const char* s, int i) {
        map<string, int> ans;         // 总表
        string name;                  // 当前正在收集的原子名
        map<string, int> pre;         // 子公式的 map
        bool hasPre = false;          // 用于区分 pre 是否存在
        int cnt = 0;                  // 倍数
        int n = strlen(s);

        while (i < n && s[i] != ')') {
            if ((s[i] >= 'A' && s[i] <= 'Z') || s[i] == '(') {
                fill(ans, name, pre, hasPre, cnt);
                name.clear();
                pre.clear();
                hasPre = false;
                cnt = 0;

                if (s[i] >= 'A' && s[i] <= 'Z') {
                    name.push_back(s[i++]);
                } else { // '('
                    pre = f(s, i + 1);
                    hasPre = true;
                    i = where + 1;
                }
            } else if (s[i] >= 'a' && s[i] <= 'z') {
                name.push_back(s[i++]);
            } else if (isdigit(s[i])) {
                cnt = cnt * 10 + (s[i++] - '0');
            }
        }

        fill(ans, name, pre, hasPre, cnt);
        where = i;  // 更新全局 where
        return ans;
    }

    void fill(map<string, int>& ans, string& name, map<string, int>& pre, bool& hasPre, int cnt) {
        if (!name.empty() || hasPre) {
            cnt = cnt == 0 ? 1 : cnt;
            if (!name.empty()) {
                ans[name] += cnt;
            } else {
                for (auto &p : pre) {
                    ans[p.first] += p.second * cnt;
                }
            }
        }
    }
};

```
